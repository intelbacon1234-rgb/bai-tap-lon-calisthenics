Trang Debug (Sửa lỗi Code)

Mô phỏng Input Mật khẩu thực tế
1. Đoạn mã lỗi ban đầu (Đề bài cung cấp)

*Lưu ý: Đoạn mã này bị lỗi logic ở câu lệnh điều kiện if khiến chức năng hoạt động sai hoàn toàn.* 


function togglePass() {
    let passField = document.getElementById("password");
    
    // LỖI NẰM Ở ĐÂY: Sử dụng dấu '=' đơn là phép gán, không phải phép so sánh
    if (passField.type = "password") { 
        passField.type = "text";
    } else {
        passField.type = "password";
    }
}
2. Đoạn mã đã sửa lỗi hoàn chỉnh (Kèm giải thích chi tiết) 

// Định nghĩa hàm xử lý ẩn/hiện mật khẩu
function togglePass() {
    // Bước 1: Tìm và lấy ra thẻ input mật khẩu bằng ID 'password'
    let passField = document.getElementById("password");
    
    // Bước 2: Kiểm tra nếu phần tử tồn tại trên trang thì mới xử lý tiếp
    if (passField) {
        
        // ĐÃ SỬA LỖI: Thay thế dấu '=' thành dấu '===' (Toán tử so sánh nghiêm ngặt)
        // GIẢI THÍCH: 
        // - Lỗi cũ (passField.type = "password") vô tình biến câu lệnh thành phép GÁN giá trị "password" vào thuộc tính type. Phép gán này luôn trả về giá trị "truthy", khiến khối lệnh IF luôn luôn chạy và đổi thành dạng "text", khối ELSE không bao giờ được lọt vào.
        // - Mã mới dùng '===' giúp so sánh chính xác xem type hiện tại ĐANG LÀ "password" hay không để chuyển đổi qua lại.
        
        if (passField.type === "password") {
            // Nếu type đang là mật khẩu ẩn, tiến hành đổi thành 'text' để hiện chữ rõ ràng
            passField.type = "text";
        } else {
            // Ngược lại, nếu đang hiện chữ, tiến hành che lại bằng cách đổi type về 'password'
            passField.type = "password";
        }
    }
}
